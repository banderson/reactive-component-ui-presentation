Manage Your State 
var Transactions = React.createClass({

  // like a declarative constructor
  getInitialState: function(){
    return {
      trx: [],
      symbols: ['GOOG', 'AAPL', 'FB'],
      running: false
    }
  },

  // this receives new real-time data and updates local state
  _handleTransaction: function(data) {
    var transactions = this.state.trx.concat(data);
    if (transactions.length > 10) {
      transactions.shift();
    }
    this.setState({trx: transactions});
  }

  // Not shown: replaceState(), forceUpdate()...
});
